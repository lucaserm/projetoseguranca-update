CREATE TABLE Responsaveis(
    telefone_responsavel varchar(14),
    email_responsavel varchar(50),
    nome_responsavel varchar(50),
    id serial PRIMARY KEY
);
CREATE TABLE Cursos(
    nome_curso varchar(30),
    periodo_cursos varchar(20),
    id serial PRIMARY KEY
);
CREATE TABLE Estudantes(
    nome_estudante varchar(50),
    email_institucional varchar(50),
    cpf varchar(14),
    ra varchar(6),
    foto varchar,
    id_responsaveis int references Responsaveis(id),
    id serial PRIMARY KEY
);
CREATE TABLE Advertencias(
    descricao varchar,
    data_advertencia date,
    id_estudantes int references Estudantes(id),
    id serial PRIMARY KEY
);
CREATE TABLE Registros(
    dia_hora_saida date,
    descricao varchar,
    id_estudantes int references Estudantes(id),
    id serial PRIMARY KEY
);
CREATE TABLE Disciplinas(
    nome_disciplina varchar(30),
    turma varchar(10),
    semestre int,
    id_cursos int references Cursos(id),
    id serial PRIMARY KEY
);
CREATE TABLE Matricula(
    ano_matricula int,
    id_estudantes int references Estudantes(id),
    id_cursos int references Cursos(id),
    id serial PRIMARY KEY
);
CREATE TABLE Horarios(
    dia_semana varchar(20),
    periodo_horarios varchar(20),
    tempo_inicio int,
    tempo_fim int
    id_disciplinas int references Disciplinas(id),
    id serial PRIMARY KEY
);
CREATE TABLE HorariosEstudantes(
    id_horarios int references Horarios(id),
    id_estudantes int references Estudantes(id),
    id serial PRIMARY KEY
);
